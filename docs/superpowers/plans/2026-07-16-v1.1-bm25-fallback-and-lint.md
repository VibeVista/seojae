**Goal:** Ship v1.1: a zero-heavy-deps BM25 search fallback inside `tools/search.py` and a deterministic wiki linter `tools/lint.py`, per the approved spec `docs/superpowers/specs/2026-07-16-v1.1-bm25-fallback-and-lint-design.md`.

**Architecture:** BM25 lives in a new pure-Python module `tools/bm25.py`; `tools/search.py` gains a `--backend {auto,chroma,bm25}` flag with per-marker index detection (chroma → `search-index/chroma.sqlite3`, BM25 → `search-index/bm25.json`) and dispatches to either backend behind the existing CLI contract. `tools/lint.py` is a standalone stdlib+PyYAML checker with 5 deterministic checks, reusing `parse_frontmatter`/`_should_index` from `tools/search.py` but with its own wikilink extractor.

**Tech Stack:** Python 3.9+ stdlib + PyYAML (lite tier); chromadb + sentence-transformers (full tier, unchanged); pytest; GitHub Actions.

## Global Constraints

- Both new tools must run on **stdlib + PyYAML only** — no new packages in either requirements file (spec "Error handling").
- BM25 params: **k1=1.5, b=0.75**; IDF = `log(1 + (N - n + 0.5) / (n + 0.5))` (non-negative, Lucene-style).
- Output format for query results must be byte-identical to chroma's: `{path} [score: {score:.2f}]`.
- Exit codes everywhere: `0` success, `1` error, `2` index missing (search) / usage error (lint).
- `--print-model` under chroma must keep today's exact two lines (`backend=search-chromadb` + `model=...`) — asserted by tests and quoted in `docs/connected-wikis-guide.ko.md`.
- Multi-collection flags (`--collections`, `--wiki-root`, non-default `--collection`) are chroma-only; the `--collection` guard must compare against the default `COLLECTION_NAME` (`"wiki"`), never mere presence.
- The BM25 CLI paths must never import `sentence_transformers` or `chromadb`.
- WIKI_SCHEMA `schema_version` stays `"1.0"`.

```python
# tools/bm25.py
"""Pure-Python BM25 Okapi backend for Seojae search.

No dependencies beyond the stdlib — this module must stay importable
under the lite install (requirements-lite.txt).
"""
from __future__ import annotations

import json
import math
import re
from pathlib import Path

K1 = 1.5
B = 0.75

_WORD_RE = re.compile(r"[a-z0-9_]+")
# Hangul jamo/syllables, Hiragana/Katakana, CJK ideographs
_CJK_RE = re.compile(
    r"[ᄀ-ᇿ぀-ヿ㄰-㆏一-鿿가-힯]+"
)


def tokenize(text: str) -> list[str]:
    """Lowercase Latin/numeric word tokens + CJK character bigrams."""
    text = text.lower()
    tokens = _WORD_RE.findall(text)
    for run in _CJK_RE.findall(text):
        if len(run) == 1:
            tokens.append(run)
        else:
            tokens.extend(run[i:i + 2] for i in range(len(run) - 1))
    return tokens
```

```python
class BM25IndexError(Exception):
    """Index file missing or unreadable — caller maps this to exit code 2."""


class BM25Index:
    """BM25 Okapi over a JSON-persisted corpus.

    IDF and avgdl are derived at query time from the stored per-document
    data — never cached — so upserts stay consistent without a rebuild.
    """

    def __init__(self, docs: dict | None = None) -> None:
        # docs: {doc_id: {"tf": {token: count}, "len": int}}
        self.docs = docs or {}

    def __len__(self) -> int:
        return len(self.docs)

    @classmethod
    def load(cls, path: str | Path) -> "BM25Index":
        p = Path(path)
        try:
            data = json.loads(p.read_text(encoding="utf-8"))
            docs = data["docs"]
            if not isinstance(docs, dict):
                raise TypeError("docs must be a dict")
        except (OSError, ValueError, KeyError, TypeError) as e:
            raise BM25IndexError(f"cannot load BM25 index at {p}: {e}") from e
        return cls(docs)

    def save(self, path: str | Path) -> None:
        p = Path(path)
        p.parent.mkdir(parents=True, exist_ok=True)
        p.write_text(
            json.dumps({"version": 1, "docs": self.docs}, ensure_ascii=False),
            encoding="utf-8",
        )

    def upsert(self, doc_id: str, text: str, title: str = "") -> None:
        tokens = tokenize(text)
        tf: dict[str, int] = {}
        for t in tokens:
            tf[t] = tf.get(t, 0) + 1
        self.docs[doc_id] = {"tf": tf, "len": len(tokens), "title": title}

    def query(self, q: str, top_n: int = 5) -> list[tuple[str, float]]:
        n_docs = len(self.docs)
        if n_docs == 0:
            return []
        q_tokens = tokenize(q)
        if not q_tokens:
            return []
        avgdl = sum(d["len"] for d in self.docs.values()) / n_docs

        scores: dict[str, float] = {}
        for term in set(q_tokens):
            df = sum(1 for d in self.docs.values() if term in d["tf"])
            if df == 0:
                continue
            idf = math.log(1 + (n_docs - df + 0.5) / (df + 0.5))
            q_freq = q_tokens.count(term)
            for doc_id, d in self.docs.items():
                tf = d["tf"].get(term, 0)
                if tf == 0:
                    continue
                denom = tf + K1 * (1 - B + B * d["len"] / avgdl) if avgdl else tf + K1
                scores[doc_id] = scores.get(doc_id, 0.0) + (
                    q_freq * idf * tf * (K1 + 1) / denom
                )

        ranked = sorted(
            ((doc_id, s) for doc_id, s in scores.items() if s > 0),
            key=lambda x: (-x[1], x[0]),
        )[:top_n]
        if not ranked:
            return []
        top_score = ranked[0][1]
        return [(doc_id, s / top_score) for doc_id, s in ranked]
```

```python
# tools/lint.py
"""Deterministic wiki health checks (no LLM). stdlib + PyYAML only.

Exit codes: 0 clean, 1 issues found, 2 usage error.
"""
from __future__ import annotations

import argparse
import json
import re
import sys
from dataclasses import dataclass
from pathlib import Path

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from tools.search import _should_index, parse_frontmatter  # noqa: E402

_WIKILINK_RE = re.compile(r"\[\[([^\]]+)\]\]")


def extract_link_targets(text: str) -> list[str]:
    """Page-name targets of wikilinks.

    NOT clean_wikilinks(): that helper keeps the *display* text of
    [[Page|Display]] for embedding readability. Lint needs the page name.
    """
    targets = []
    for inner in _WIKILINK_RE.findall(text):
        target = inner.split("|", 1)[0].split("#", 1)[0].strip()
        if target:
            targets.append(target)
    return targets


@dataclass
class Page:
    path: Path
    stem: str
    fm: dict
    body: str
    fm_error: str | None


@dataclass
class Issue:
    check: str
    path: str
    message: str
    warning: bool = False


def load_pages(wiki_path: Path) -> list[Page]:
    pages = []
    for p in sorted(wiki_path.rglob("*.md")):
        if not _should_index(p, wiki_path):
            continue
        try:
            text = p.read_text(encoding="utf-8")
        except OSError as e:
            pages.append(Page(p, p.stem, {}, "", f"unreadable: {e}"))
            continue
        fm, body = parse_frontmatter(text)
        err = None if fm else "missing or unparseable frontmatter"
        pages.append(Page(p, p.stem, fm, body, err))
    return pages
```

```python
import datetime

_VALID_TYPES = {"entity", "concept", "source", "synthesis"}
_DATE_RE = re.compile(r"^\d{4}-\d{2}-\d{2}$")
_REQUIRED_FIELDS = ("title", "type", "created", "updated")
_LIST_FIELDS = ("tags", "aliases", "sources")


def _valid_date(v) -> bool:
    if isinstance(v, datetime.date):
        return True
    return isinstance(v, str) and bool(_DATE_RE.match(v))


def _page_names(page: Page) -> set[str]:
    """All lowercase names a wikilink may resolve to for this page."""
    names = {page.stem.lower()}
    title = page.fm.get("title")
    if isinstance(title, str) and title.strip():
        names.add(title.strip().lower())
    aliases = page.fm.get("aliases")
    if isinstance(aliases, list):
        names.update(str(a).strip().lower() for a in aliases if str(a).strip())
    return names


def run_checks(pages: list[Page], index_md_text: str) -> list[Issue]:
    issues: list[Issue] = []
    name_to_page: dict[str, Page] = {}
    for page in pages:
        for name in _page_names(page):
            name_to_page.setdefault(name, page)

    # 3. frontmatter validity (run first: informs the rest but never aborts)
    for page in pages:
        rel = str(page.path)
        if page.fm_error:
            issues.append(Issue("frontmatter", rel, page.fm_error))
            continue
        for f in _REQUIRED_FIELDS:
            if f not in page.fm:
                issues.append(Issue("frontmatter", rel, f"missing required field: {f}"))
        t = page.fm.get("type")
        if t is not None and t not in _VALID_TYPES:
            issues.append(Issue("frontmatter", rel,
                                f"invalid type: {t!r} (must be one of {sorted(_VALID_TYPES)})"))
        for f in ("created", "updated"):
            if f in page.fm and not _valid_date(page.fm[f]):
                issues.append(Issue("frontmatter", rel,
                                    f"{f} must be YYYY-MM-DD, got {page.fm[f]!r}"))
        for f in _LIST_FIELDS:
            if f in page.fm and not isinstance(page.fm[f], list):
                issues.append(Issue("frontmatter", rel, f"{f} must be a list"))

    # 1. broken wikilinks + 2. orphan inbound counting (single pass)
    inbound: dict[Path, int] = {p.path: 0 for p in pages}
    for page in pages:
        for target in extract_link_targets(page.body):
            resolved = name_to_page.get(target.strip().lower())
            if resolved is None:
                issues.append(Issue("broken-link", str(page.path),
                                    f"[[{target}]] does not match any page title, alias, or filename"))
            elif resolved.path != page.path:
                inbound[resolved.path] += 1

    # 2. orphans (warnings; index.md listing does not count as a link)
    for page in pages:
        if inbound[page.path] == 0:
            issues.append(Issue("orphan", str(page.path),
                                "no inbound wikilinks from other pages", warning=True))

    # 4. filename uniqueness across wiki/ subdirectories
    stems: dict[str, list[Page]] = {}
    for page in pages:
        stems.setdefault(page.stem.lower(), []).append(page)
    for stem, group in stems.items():
        if len(group) > 1:
            paths = ", ".join(str(p.path) for p in group)
            issues.append(Issue("duplicate-filename", str(group[0].path),
                                f"stem '{stem}' used by multiple pages: {paths}"))

    # 5. index.md sync
    index_targets = {t.strip().lower() for t in extract_link_targets(index_md_text)}
    for page in pages:
        if not (_page_names(page) & index_targets):
            issues.append(Issue("index-sync", str(page.path),
                                f"page not listed in index.md: {page.stem}"))
    for target in sorted(index_targets - set(name_to_page)):
        issues.append(Issue("index-sync", "index.md",
                            f"index entry [[{target}]] points at no existing page"))

    return issues
```

Note: the `index-sync` "points at no existing page" message must interpolate the original-case target — store `extract_link_targets(index_md_text)` once, and keep a lowercase→original mapping so `Ghost Entry` (not `ghost entry`) appears in the message:

```python
        issues.append(Issue("index-sync", "index.md",
```

```python
_WIKI_DEFAULT = Path(__file__).resolve().parent.parent / "wiki"
_INDEX_DEFAULT = Path(__file__).resolve().parent.parent / "index.md"


def main() -> None:
    parser = argparse.ArgumentParser(description="Seojae deterministic wiki lint")
    parser.add_argument("--wiki-path", default=str(_WIKI_DEFAULT), metavar="PATH")
    parser.add_argument("--index-md", default=str(_INDEX_DEFAULT), metavar="FILE",
                        help="index.md catalog file (not the search-index directory)")
    parser.add_argument("--json", action="store_true", help="machine-readable output")
    parser.add_argument("--strict", action="store_true",
                        help="orphan warnings also fail (exit 1)")
    args = parser.parse_args()

    wiki_path = Path(args.wiki_path)
    index_md = Path(args.index_md)
    if not wiki_path.is_dir():
        print(f"Error: wiki path not found: {wiki_path}", file=sys.stderr)
        sys.exit(2)
    if not index_md.is_file():
        print(f"Error: index file not found: {index_md}", file=sys.stderr)
        sys.exit(2)

    pages = load_pages(wiki_path)
    issues = run_checks(pages, index_md.read_text(encoding="utf-8"))

    if args.json:
        print(json.dumps(
            [{"check": i.check, "path": i.path, "message": i.message} for i in issues],
            ensure_ascii=False, indent=2))
    else:
        by_check: dict[str, list[Issue]] = {}
        for i in issues:
            by_check.setdefault(i.check, []).append(i)
        for check in sorted(by_check):
            print(f"[{check}]")
            for i in by_check[check]:
                tag = " (warning)" if i.warning else ""
                print(f"  {i.path}: {i.message}{tag}")
        print(f"{len(pages)} pages checked, "
              f"{sum(1 for i in issues if not i.warning)} issues, "
              f"{sum(1 for i in issues if i.warning)} warnings")

    failing = [i for i in issues if not i.warning or args.strict]
    sys.exit(1 if failing else 0)


if __name__ == "__main__":
    main()
```
